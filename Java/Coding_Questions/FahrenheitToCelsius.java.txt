package importantQuestions;

import java.util.Scanner;

public class FahrenheitToCelsiusProgram {
	public static void main(String[] args) {
		//Formula to calculate Fahrenhit to Celsius C = [ (°F-32)×5]/9
		float fahrenheit;
		Scanner sc= new Scanner(System.in);
		System.out.println("Enter the Fahrenheit Temperature ");
		fahrenheit= sc.nextInt();
		float celsius= (fahrenheit-32)*5/9;
		System.out.println("Here is the Celsius Converted: "+celsius);
		 
	}
}


Output:
Enter the Fahrenheit Temperature 
98
Here is the Celsius Converted: 36.666668
